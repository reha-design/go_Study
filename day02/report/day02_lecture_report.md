# Go 강의 보고서
## 1. 강의 목표 및 성취 수준

본 강의는 Go 언어의 실수 표현 방식의 특징과 오차 해결법, 표준 출력을 위한 다양한 서식 지정자 사용법, 사칙연산 및 비트/비교 연산의 원리를 이해하고, 프로그램 구조화와 중복 제거를 위한 함수의 정의 및 활용을 직접 코드로 작성·실행해 보는 것을 목표로 진행하였습니다.

수강생들은 강의와 실습을 통해 다음 내용을 수행할 수 있도록 학습하였습니다.

1. float32와 float64의 정밀도 차이를 이해하고, 컴퓨터가 실수를 다룰 때 발생하는 오차의 원인과 3가지 해결책(반올림 출력, 엡실론 비교, 정수 변환 연산)을 적용할 수 있다.
2. fmt.Printf를 사용하여 데이터 타입에 맞는 서식 문자(%v, %T, %d, %f 등)를 지정하고 줄 정렬 및 0 채우기와 같은 정렬 포맷팅을 수행할 수 있다.
3. 사칙연산, 나눗셈에서의 정수/실수 연산 차이, 나머지 연산자(%)의 부호 규칙 및 복합 대입/증감 연산자의 올바른 사용법을 이해할 수 있다.
4. 비트 연산자(&, |, ^, &^, <<, >>)의 동작 원리를 2진수 관점에서 이해하고 실습할 수 있다.
5. 비교 연산자를 통한 대소/동등 판별 및 문자열 사전식 비교 방식의 원리와 실수 비교 시 주의점을 설명할 수 있다.
6. 함수의 매개변수와 반환값 구조를 이해하고, Go 특유의 다중 반환값과 Named Return Value(이름이 명명된 반환값)를 사용해 중복 없고 가독성 높은 코드를 작성할 수 있다.

#### ① `06_float_precision/main.go`

```go
package main

import "fmt"

func main() {
	// [1. 실수 타입: float32 vs float64]
	// 실수는 소수점이 있는 숫자입니다.
	// float64가 float32보다 더 넓은 공간을 써서 더 정밀하게 숫자를 보관합니다.
	// 그래서 Go에서는 특별한 이유가 없다면 float64를 기본으로 씁니다.
	var f1 float32 = 0.1234567890123456789
	var f2 float64 = 0.1234567890123456789

	fmt.Println("1. 정밀도 비교 (누가 더 정확할까?):")
	fmt.Println("   float32 (작은 바구니):", f1) // 소수점 아래 약 7자리 이후로는 값이 틀어짐
	fmt.Println("   float64 (큰 바구니):", f2)   // 소수점 아래 약 15자리까지 정확

	// [2. 지수 표기법: e 사용하기]
	// 아주 큰 숫자(0이 많은 숫자)나 아주 작은 숫자를 짧게 쓸 때 사용합니다.
	var e1 float64 = 1.23e4  // 1.23 * 10,000 = 12300
	var e2 float64 = 1.23e-2 // 1.23 * 0.01 = 0.0123

	fmt.Println("\n2. 지수 표기법 (큰 수/작은 수 표현):")
	fmt.Printf("   1.23e4  (1.23만) : %v\n", e1)
	fmt.Printf("   1.23e-2 (0.0123) : %v\n", e2)

	// [3. 소수점 자리수 직접 지정하기]
	// 출력할 때 "나는 소수점 2자리까지만 보여줘!"라고 정할 수 있습니다.
	fmt.Println("\n3. 예쁘게 출력하기 (포맷팅):")
	fmt.Printf("   소수점 2자리까지: %.2f\n", f2) // .2f -> 소수점 아래 2자리

	// [4. 매우 중요: 컴퓨터는 실수를 완벽히 계산 못 함]
	// 컴퓨터는 모든 숫자를 0과 1(이진수)로 생각합니다. 
	// 우리가 쓰는 0.1이나 0.2를 이진수로 바꾸면 무한소수가 되어버려 끝부분이 아주 조금씩 잘립니다.
	fmt.Println("\n4. 실수의 계산 오차 (컴퓨터의 한계):")
	var a float64 = 0.1
	var b float64 = 0.2
	fmt.Printf("   0.1 + 0.2 = %v\n", a+b) 
	// 결과: 0.30000000000000004 (0.3이 아닙니다!)

	// [5. 오차를 해결하는 세 가지 요령]
	fmt.Println("\n5. 이 오차를 어떻게 피할까?")

	// 방법 1: 그냥 반올림해서 보여주기 (눈속임)
	// - 보여줄 때만 소수점 자리를 끊어서 오차를 안 보이게 합니다.
	fmt.Printf("   [방법1] %.1f + %.1f = %.1f  (화면에만 0.3으로 출력)\n", a, b, a+b)

	// 방법 2: 엡실론(Epsilon) 비교
	// - "두 값의 차이가 아주아주 작으면(예: 0.000000001보다 작으면) 같은 값으로 치자!"는 약속입니다.
	const epsilon = 1e-9 
	expected := 0.3
	diff := (a + b) - expected
	
	// diff(차이값)가 음수면 양수로 바꿔주는 작업 (절댓값)
	if diff < 0 {
		diff = -diff 
	}

	if diff < epsilon {
		fmt.Println("   [방법2] 차이가 거의 없으므로 0.3으로 인정!")
	}

	// 방법 3: 정수로 바꿔서 계산하기 (가장 정확!)
	// - 0.1과 0.2를 10배씩 곱해서 정수 1과 2로 만듭니다.
	// - 정수끼리는 오차 없이 완벽하게 더해집니다 (1+2=3).
	// - 마지막에 다시 10으로 나눠서 0.3을 만듭니다. (돈 계산할 때 주로 쓰는 방식)
	ai := int64(a*10 + 0.5) // 반올림 효과를 위해 0.5를 더함
	bi := int64(b*10 + 0.5) 
	result := float64(ai+bi) / 10 
	fmt.Printf("   [방법3] 정수로 변신시켜 계산: %v\n", result)
}
```

#### ② `07_fmt_specifiers/main.go`

```go
package main

import "fmt"

func main() {
	// 테스트용 데이터들
	var i = 123
	var f = 3.141592
	var s = "Go Language"
	var b = true

	// [1. 가장 많이 쓰는 '만능' 서식]
	fmt.Println("--- 1. 자주 쓰는 서식 문자 ---")
	// %v: value의 약자. 타입을 몰라도 알아서 적절한 형식으로 보여줍니다. (가장 많이 씀!)
	fmt.Println("값 그대로 출력 (%v):", i)
	// %T: Type의 약자. 이 변수의 바구니 종류가 무엇인지 보여줍니다.
	fmt.Println("데이터 타입 출력 (%T):", i)
	
	// %v와 %T만 알아도 웬만한 출력은 다 해결할 수 있습니다.
	fmt.Printf("값: %v, 타입: %T\n", s, s)

	// [2. 정수(Integer)를 위한 서식]
	fmt.Println("\n--- 2. 정수 (숫자) 전용 ---")
	// %d: decimal(10진수)의 약자. 우리가 평소 쓰는 숫자를 출력합니다.
	fmt.Printf("10진수(%%d): %d\n", i)
	// %b: binary(2진수)의 약자. 컴퓨터가 이해하는 0과 1로 보여줍니다.
	fmt.Printf(" 2진수(%%b): %b\n", i)
	// %x: hex(16진수)의 약자. 메모리 주소 등을 표현할 때 씁니다.
	fmt.Printf("16진수(%%x): %x\n", i)

	// [3. 실수(Float)를 위한 서식]
	fmt.Println("\n--- 3. 실수 (소수점) 전용 ---")
	// %f: float의 약자. 소수점 아래 6자리까지 기본으로 보여줍니다.
	fmt.Printf("기본 출력(%%f): %f\n", f)
	// %.2f: 마침표(.) 뒤의 숫자로 소수점 몇 자리까지 보여줄지 정합니다. (중요!)
	fmt.Printf("소수점 2자리(%%.2f): %.2f\n", f)

	// [4. 문자열 및 불리언 서식]
	fmt.Println("\n--- 4. 문자열 및 참/거짓 ---")
	// %s: string의 약자. 글자 그대로 출력합니다.
	fmt.Printf("문자열(%%s): %s\n", s)
	// %q: quote의 약자. 따옴표를 포함해서 출력해줍니다. (구분하기 좋음)
	fmt.Printf("따옴표 포함(%%q): %q\n", s)
	// %t: true/false의 약자. 불리언 값을 출력합니다.
	fmt.Printf("불리언(%%t): %t\n", b)

	// [5. 줄 맞춤 기능]
	fmt.Println("\n--- 5. 정렬 및 패딩 ---")
	fmt.Printf("|%5d| (5칸 공간 확보 후 우측 정렬)\n", i)
	fmt.Printf("|%-5d| (5칸 공간 확보 후 좌측 정렬)\n", i)
	fmt.Printf("|%05d| (5칸 공간 확보 후 빈 곳은 0으로 채우기)\n", i)

	// [6. 특수 문자 (제어 문자)]
	fmt.Println("\n--- 6. 특수 기능 (이스케이프 문자) ---")
	fmt.Printf("줄바꿈: \\n 은 엔터키를 친 것과 같습니다.\n")
	fmt.Printf("탭 문자: \\t 는 \t 탭키를 누른 것과 같습니다.\n")
	fmt.Printf("큰따옴표: \\\" 는 글자 안에서 \"따옴표\"를 쓰고 싶을 때 씁니다.\n")
}
```

#### ③ `08_arithmetic/main.go`

```go
package main

import "fmt"

func main() {
	// [1. 기본 산술 연산자: 사칙연산]
	a, b := 17, 5

	fmt.Println("1. 기본 산술 연산 (a=17, b=5):")
	fmt.Println("   a + b =", a+b) 
	fmt.Println("   a - b =", a-b) 
	fmt.Println("   a * b =", a*b) 
	
	// [중요!] 정수와 정수를 나누면 결과도 무조건 '정수'입니다.
	// 17 나누기 5는 3.4지만, 소수점은 무참히 버려지고 3만 남습니다.
	fmt.Println("   a / b =", a/b)  // 결과: 3

	// [% 연산자]: '나머지'를 구합니다. 17을 5로 나눈 뒤 남은 2를 알려줍니다.
	// 주로 '홀수/짝수 판별'이나 '배수 확인'에 아주 많이 씁니다.
	fmt.Println("   a % b =", a%b)  // 결과: 2

	// [2. 실수 결과가 필요할 때]
	fmt.Println("\n2. 소수점 결과가 보고 싶을 때:")
	// 나누기 전에 숫자를 실수(float64)로 먼저 변신시켜야 소수점이 살아납니다.
	fmt.Printf("   float64(a)/float64(b) = %f\n", float64(a)/float64(b))

	// [3. 나머지 연산자의 부호]
	// 나머지의 부호는 무조건 '앞에 있는 숫자'를 따라갑니다.
	fmt.Println("\n3. 나머지(%) 부호 규칙:")
	fmt.Printf("   -7 %% 3 = %d (-7의 부호를 따름)\n", -7%3)

	// [4. 복합 대입 연산자: 줄여 쓰기]
	// x = x + 3 처럼 자기 자신에게 다시 값을 넣을 때 짧게 줄여 쓸 수 있습니다.
	fmt.Println("\n4. 복합 대입 연산자 (줄여 쓰기):")
	x := 10
	x += 3 // x = x + 3 과 같은 뜻입니다. (결과: 13)
	fmt.Printf("   x += 3  → x = %d\n", x)
	x *= 2 // x = x * 2 와 같은 뜻입니다. (결과: 26)
	fmt.Printf("   x *= 2  → x = %d\n", x)

	// [5. 증감 연산자: 하나씩 올리기/내리기]
	// n++: n에 1을 더합니다. / n--: n에서 1을 뺍니다.
	// 게임의 '레벨업'이나 '점수 추가' 등에서 아주 유용합니다.
	fmt.Println("\n5. 증감 연산자 (++ / --):")
	n := 5
	n++ // 5에서 1이 늘어나 6이 됨
	fmt.Printf("   n++ → n = %d\n", n)
	n-- // 6에서 다시 1이 줄어 5가 됨
	fmt.Printf("   n-- → n = %d\n", n)

	// [6. 연산자 우선순위]
	// 우리가 수학 시간에 배운 것처럼 곱하기(*), 나누기(/)가 더하기(+), 빼기(-)보다 먼저 계산됩니다.
	// 헷갈릴 때는 무조건 괄호 ( )를 쓰면 안전합니다!
	fmt.Println("\n6. 연산자 우선순위 (수학과 동일):")
	fmt.Printf("   2 + 3*4    = %d (3*4를 먼저 계산함)\n", 2+3*4)
	fmt.Printf("   (2+3)*4   = %d (괄호를 먼저 계산함)\n", (2+3)*4)
}
```

#### ④ `09_bitwise/main.go`

```go
package main

import "fmt"

func main() {
	// [비트 연산자: 컴퓨터의 언어로 대화하기]
	// 비트 연산은 숫자를 0과 1의 조합(2진수)으로 보고 계산하는 아주 미세한 작업입니다.
	// ※ 비전공자 Tip: 지금은 "이런 게 있구나" 하고 구경만 하셔도 충분합니다! 나중에 정말 필요할 때 다시 봐도 늦지 않아요.
	
	var a int = 12 // 2진수로 1100
	var b int = 10 // 2진수로 1010

	fmt.Println("--- 비트 연산자 (2진수 계산기) ---")
	fmt.Printf("a = %d \t (2진수: %04b)\n", a, a)
	fmt.Printf("b = %d \t (2진수: %04b)\n", b, b)

	fmt.Println("\n--- 1. 비트 AND (&) ---")
	// 둘 다 1일 때만 1을 줍니다. (까다로운 조건)
	// 1100 & 1010 = 1000 (8)
	fmt.Printf("a & b \t = %d \t (둘 다 1인 곳만 1)\n", a&b)

	fmt.Println("\n--- 2. 비트 OR (|) ---")
	// 둘 중 하나라도 1이면 1을 줍니다. (너그러운 조건)
	// 1100 | 1010 = 1110 (14)
	fmt.Printf("a | b \t = %d \t (하나라도 1이면 1)\n", a|b)

	fmt.Println("\n--- 3. 비트 XOR (^) ---")
	// 서로 다를 때만 1을 줍니다. (청개구리 조건)
	// 1100 ^ 1010 = 0110 (6)
	fmt.Printf("a ^ b \t = %d \t (서로 다를 때만 1)\n", a^b)

	fmt.Println("\n--- 4. 비트 클리어 (&^) ---")
	// a 비트 중에서 b 비트가 1인 자리를 0으로 청소해버립니다.
	// 1100 &^ 1010 = 0100 (4)
	fmt.Printf("a &^ b \t = %d \t (b가 1인 자리를 0으로)\n", a&^b)

	fmt.Println("\n--- 5. 비트 시프트 (<<, >>) ---")
	// 비트들을 왼쪽이나 오른쪽으로 밀어버립니다.
	var c int = 1 // 0001
	
	// << 2 : 왼쪽으로 2칸 밀기 (0001 -> 0100)
	// 왼쪽으로 한 칸 밀 때마다 숫자가 2배씩 커집니다.
	fmt.Printf("1 << 2 \t = %d \t (왼쪽으로 2칸 밀기)\n", c<<2)
	
	var d int = 8 // 1000
	// >> 2 : 오른쪽으로 2칸 밀기 (1000 -> 0010)
```

## 3. 핵심 교육 내용 요약

### 실수의 특징과 정밀도 오차

Go의 두 가지 실수 타입인 `float32`와 `float64`의 공간적 차이 및 정밀도 검증을 진행하였습니다. 소수점 이하 약 7자리까지 표현하는 `float32`에 비해, 약 15자리까지 세밀하게 보관 가능한 `float64`를 실무 환경에서 기본적으로 채택해야 함을 확인하였습니다.

아울러 컴퓨터가 10진수 실수를 이진수 체계로 번역하고 저장하면서 꼬리가 잘려 발생하는 '계산 오차(무한소수)' 현상의 유래를 분석하고, 실무에서 이를 회피하기 위한 3가지 요령을 학습하였습니다.

- 반올림 포맷팅 출력 (시각적인 처리)
- 엡실론(Epsilon) 절대 차이 비교
- 곱연산을 활용해 임시 정수화한 뒤 복원하여 연산

### Printf 서식 지정자와 정렬

`fmt.Printf` 함수를 통하여 상황에 맞춘 데이터 형식 출력을 실습하고, 데이터별 서식 문자 종류를 익혔습니다.

- 만능 서식: `%v` (값 출력), `%T` (타입 출력)
- 정수 서식: `%d` (10진수), `%b` (2진수), `%x` (16진수)
- 실수 서식: `%f` (실수 출력), `%.2f` (소수점 자릿수 한정)
- 기타 서식: `%s` (문자열), `%q` (따옴표 동반), `%t` (불리언)
- 정렬 포맷: `%5d` (우측 정렬), `%-5d` (좌측 정렬), `%05d` (빈자리 0 패딩)

### 산술 및 복합 연산 규칙

정수형 간의 사칙연산 수행 시 소수점이 버려지는 나눗셈의 특성을 파악하고, 실수 연산 결과가 필요하다면 사전에 형변환을 가해 계산해야 함을 확인하였습니다.
또한, 짝수/홀수 판정이나 주기를 구분할 때 사용되는 나머지 연산자(`%`)가 음수 계산 시 앞에 놓인 피제수의 부호를 온전히 반영한다는 고유의 규약을 실습하였습니다.

더불어 증감 연산자(`++`, `--`)의 경우 타 언어와 달리 독립적인 한 줄 문장으로만 기능하며, 다른 출력이나 식 내부에서 동시에 중첩하여 결합할 수 없다는 Go 언어만의 엄격하고 명료한 안전 지침을 파악하였습니다.

### 비트 및 비교 연산

컴퓨터의 가장 기초적인 논리 단위인 2진수 연산 기법을 이해하고 비트 연산자를 활용하였습니다.

- 비트 AND (`&`): 양쪽 모두 1이어야 1
- 비트 OR (`|`): 최소 한쪽이 1이면 1
- 비트 XOR (`^`): 두 비트의 상태가 다르면 1
- 비트 클리어 (`&^`): 대상 비트 중 특정 열의 값을 0으로 리셋
- 비트 시프트 (`<<`, `>>`): 왼쪽/오른쪽으로 비트를 이동해 2배 증가 혹은 반감 효과 유도

대소 및 일치 여부를 파악하는 비교 연산자가 `bool` 결과를 산출함을 관찰하고, 문자열 간의 부등호 비교는 유니코드 사전식 나열 순번에 기인함을 대조 실습하였습니다.

### 함수의 구성과 코드 구조화

반복을 없애고 유지보수성을 극대화하기 위해 '함수'라는 기계를 조립하는 구조를 실습하였습니다. Go 언어 함수가 제공하는 편리한 기능들을 중심으로 학습이 진행되었습니다.

- 다중 반환값 (Multiple Return Values): 한 번의 가동으로 복수의 다른 결과(총합, 평균 등)를 쉼표로 동시에 꺼내오는 기법
- Named Return Values (이름 붙은 반환값): 결과로 넘길 변수의 유형과 식별자를 함수 헤더에 미리 규정하여, 함수 본문 종단에서 인자 없이 `return` 단독 지정을 지원하는 방식
- 중복 배제(DRY): 동일한 계산 블록을 단일 함수로 결집함으로써, 평균이나 배점 공식이 추후 갱신되더라도 함수의 일부분만 수선해 전체 대상에 한 번에 적용하는 유지보수 방식의 극적 개선 확인

## 4. 강의 평가 및 교육적 성과

이번 2일차 강의에서는 데이터를 연산하는 다채로운 방식(실수 계산 극복, 서식 출력, 산술·비트·비교 연산)에서 시작하여 이를 하나의 모듈로 아름답게 묶는 함수의 정의와 사용법에 대한 깊이 있는 학습을 이행하였습니다.

수강생들은 단순히 연산 기호의 암기에 머무르지 않고 컴퓨터 아키텍처 관점에서 왜 `0.1 + 0.2`가 정밀하게 맞아떨어지지 않는지 구조적 함의를 인지하고 실무에서 에러를 비껴가는 우회 기법을 획득하였습니다. 또한, 비전공자 교육 단계에서 거부감이 큰 비트 연산을 가볍게 둘러보며 진입 방벽을 낮췄습니다.

가장 중요한 교육 성과로는 함수가 미존재하는 구조에서 발생하는 복사-붙여넣기 기반 개발의 지저분한 패턴을 직접 체험한 뒤, 함수 적용 버전을 대조 실행함으로써 '중복 배제와 유지보수 편의성'이라는 구조적 장점을 뼛속 깊이 깨달았다는 점입니다.

종합적으로 볼 때, 수강생들은 Go 언어의 데이터 연산 방법과 함수를 통한 기본 제어·구조화 패턴을 훌륭히 완수하였으며, 연산자 활용 역량과 결합하여 3일차의 핵심 영역인 조건 제어문(if, switch) 및 반복문(for)의 실전 알고리즘을 소화할 준비를 충실히 끝마쳤습니다.
